function findNextId (){

      if (data.todos[i].id > lastId) {

function filterTodos(mok, userAuth) {
  var retArray = []

  for (var i = 0; i < mok.length; i++) {
      if (mok[i].userAuth == userAuth) {
          retArray.push(mok[i])
      }
  }
